018        /**
019         * Invokes a deep getter command, given the pattern to dive after and the original object.  If the following parameters are passed:
020         * <p/>
021         * deep = "class.name.class.name"<BR>
022         * o = new Object()
023         * <p/>
024         * The result will be: java.lang.String="java.lang.String".  The method would have done the following operations:
025         * new Object().getClass = java.lang.Object.CLASS<BR>
026         * java.lang.Object.CLASS.getName() = "java.lang.Object"<BR>
027         * "java.lang.Object".getClass() = java.lang.String.CLASS<BR>
028         * java.lang.String.CLASS.getName() = "java.lang.String"<BR>
029         *
030         * @param deep The deep string (dot separated) to dive after
031         * @param o    The original object
032         */
033    
034        public static Object deepGetter(String deep, Object o) throws NoSuchMethodException, IllegalAccessException, InvocationTargetException {
035            String[] parts = deep.split("\\.");
036            String first = parts[0];
037            Method get = o.getClass().getMethod("get" + StringUtils.firstLetterUpper(first));
038            Object got = get.invoke(o);
039    
040            if (deep.contains(".")) {
041                String remain = CollectionUtil.Join(parts, 1, ".");
042                return deepGetter(remain, got);
043            } else {
044                return got;
045            }
046        }
047    }
