077        /**
078         * Reverses a map, but maintains a list of keys associated with values, instead of
079         * maintainting only one value per key
080         * @param map
081         * @return
082         */
083        public static <K, V> Map<V, Set<K>> ReverseMapWithList(Map<K, V> map) {
084            Map<V, Set<K>> rtn = new HashMap<V, Set<K>>();
085            for (Map.Entry<K, V> kvEntry : map.entrySet()) {
086                final V value = kvEntry.getValue();
087                if (!rtn.containsKey(value)) {
088                    rtn.put(value, new HashSet<K>());
089                }
090    
091                rtn.get(value).add(kvEntry.getKey());
092            }
093            return rtn;
094        }

189            StringBuffer tmpString = new StringBuffer();
190            delimiter = StringUtils.Nz(delimiter, ",");
191    
192            for (Object aC : c) {
193                String next = "" + aC;
194                if (includeEmpty || StringUtils.is(next)) {
195                    tmpString.append(next).append(delimiter);
196                }
197            }
198    
199            return StringUtils.rtrim(tmpString.toString(), delimiter.length());

285        /**
286         * This method is used to round robin entries by ratio, evenly spreading items until ratios run out.  For example,
287         * if I have the following ratios:
288         * <p/>
289         * A: 2<BR>
290         * B: 5<BR>
291         * C: 7<BR>
292         * <p/>
293         * The method would return the follwoing list:
294         * <p/>
295         * A,B,C,A,B,C,B,C,B,C,B,C,C,C
296         *
297         * @see #SpreadOut(Map,int)
298         */
299        public static <X> List<X> HandOut(Map<X, Integer> ratios) {
300            if (ratios == null) {
301                return null;
302            }
303            int i = 0;
304            int sum = CollectionUtil.SumInteger(ratios.values());
305            List<X> items = new ArrayList<X>(sum);
306    
307            while (true) {
308                i++;
309                boolean found = false;
310                for (Map.Entry<X, Integer> entry : ratios.entrySet()) {
311                    if (entry.getValue() >= i) {
312                        items.add(entry.getKey());
313                        found = true;
314                    }
315                }
316                if (!found) {
317                    break;
318                }
319            }
320            return items;
321        }

373         * Will dive into a map as deep as it can go.  When it hits a wall, will return null;
374         */
375        public static Object MapDiver(Map diveInto, Object... keys) {
376            Object _value = diveInto;
377            int i = 0;
378            while (_value != null && i < keys.length) {
379                Object key = keys[i];
380                _value = ((Map) _value).get(key);
381                if (!(_value instanceof Map) || _value == null) {
382                    return _value;
383                }
384                i++;
385            }
386            return _value;
387        }

427         *
428         * @param map               THe map to convert
429         * @param entrySeparator    What to separate map entries with
430         * @param keyValueSeparator What to separate key/value pairs with
431         * @param ignoreNulls       Whether or not to ignore null values
432         * @return String
433         */
434        public static String MapToString(Map map, String entrySeparator, String keyValueSeparator, boolean ignoreNulls) {
435            if (map == null) {
436                return null;
437            }
438            StringBuffer tmp = new StringBuffer();
439            for (Object o : map.entrySet()) {
440                Map.Entry entry = (Map.Entry) o;
441                if (ignoreNulls || entry.getValue() != null) {
442                    tmp.append(entry.getKey())
443                            .append(keyValueSeparator)
444                            .append(entry.getValue())
445                            .append(entrySeparator);
446                }
447            }
448            return tmp.toString();
449        }

451        /**
452         * Moves an item in a list to another location in a list.
453         *
454         * @param orig The original list
455         * @param from The index to move from
456         * @param to   The index to move to
457         * @return A copy of the original list with the items in the correct order
458         */
459        public static <X> List<X> MoveItem(List<X> orig, int from, int to) {
460            if (orig == null) {
461                return null;
462            }
463            List<X> ret = new ArrayList<X>();
464            X fromItem = orig.get(from);
465            for (int i = 0; i < orig.size(); i++) {
466                if (i != from) {
467                    if (i == to && from > to) {
468                        ret.add(fromItem);
469                    }
470                    ret.add(orig.get(i));
471                    if (i == to && from < to) {
472                        ret.add(fromItem);
473                    }
474                }
475            }
476            return ret;
477        }

724        /**
725         * reverse the order of a map
726         * @param map map
727         * @return reversed order map
728         */
729        public static <K, V> Map<K, V> ReverseMapOrder(Map<K, V> map) {
730            List<K> keys = new ArrayList<K>(map.keySet());
731            Map<K, V> reversed = new LinkedHashMap<K, V>();
732            for (ListIterator<K> iterator = keys.listIterator(keys.size()); iterator.hasPrevious();) {
733                K k = iterator.previous();
734                reversed.put(k, map.get(k));

831        /**
832         * This method will round robin items, keeping in line with ratios. For example,
833         * if I have the following ratios:
834         * <p/>
835         * A: 2<BR>
836         * B: 4<BR>
837         * C: 6<BR>
838         * <p/>
839         * The method would return the follwoing list:
840         * <p/>
841         * <p/>
842         * C,C,C,B,B,A,C,C,C,B,B,A
843         *
844         * @param total The number of items to return in the list
845         */
846        public static <X> List<X> SpreadOut(Map<X, Integer> ratios, int total) {
847            CountMap<X> map = new CountMap<X>();
848            List<X> ret = new ArrayList<X>();
849            for (int i = 0; i < total; i++) {
850                int min = Integer.MAX_VALUE;
851                X minX = null;
852    
853                for (X x : ratios.keySet()) {
854                    int currRatio = map.getRatio(x);
855                    int diff = NumberUtils.getPercentDiff(currRatio, ratios.get(x));
856                    if (diff < min) {
857                        minX = x;
858                        min = diff;
859                    }
860                }
861    
862                map.add(minX);
863                ret.add(minX);
864            }
865    
866            return ret;
867        }
