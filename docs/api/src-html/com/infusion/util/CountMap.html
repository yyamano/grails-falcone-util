001    package com.infusion.util;
002    
003    
004    import java.util.HashMap;
005    import java.util.Map;
006    import java.util.Set;
007    
008    /**
009     * Lets you easily maintain counts in a map.
010     *
011     * @author eric
012     */
013    public class CountMap<T> {
014    // ========================================================================================================================
015    //    Instance Fields
016    // ========================================================================================================================
017    
018        /**
019         * The key is the object you are counting, hte value is the count
020         */
021        private HashMap<T, Integer> counts = new HashMap<T, Integer>();
022    
023        /**
024         * The total counts (of everything)
025         */
026        private int total = 0;
027    
028    // ========================================================================================================================
029    //    Public Instance Methods
030    // ========================================================================================================================
031    
032        /**
033         * Increments the counter for a given item
034         */
035        public int add(T item) {
036            if (!counts.containsKey(item)) {
037                counts.put(item, 1);
038            } else {
039                counts.put(item, counts.get(item) + 1);
040            }
041            total++;
042            return counts.get(item);
043        }
044    
045        /**
046         * Returns the count for a given item
047         */
048        public Integer getCount(T o) {
049            return counts.containsKey(o) ? counts.get(o) : 0;
050        }
051    
052        /**
053         * Returns the ratio of the count for a given item.
054         */
055        public int getRatio(T item) {
056            return 100 * getCount(item) / Math.max(1, total);
057        }
058    
059        /**
060         * Returns an iterator of the items
061         */
062        public Set<Map.Entry<T, Integer>> iterator() {
063            return counts.entrySet();
064        }
065    
066    }
