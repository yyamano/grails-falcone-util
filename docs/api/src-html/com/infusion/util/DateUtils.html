228        /**
229         * Compares 2 cal1 and cal2 based on the comparator passed in.
230         * comparator can be one of the following strings: "<=", "==", ">=", "<", ">";
231         * If comparator is any other string, false is returned.
232         * If either cal1 OR cal2 is null, false is returned.
233         */
234        public static boolean compare(Calendar cal1, String comparator, Calendar cal2) {
235            if (cal1 == null || cal2 == null) {
236                return false;
237            }
238    
239            if ("==".equals(comparator)) {
240                return cal1.compareTo(cal2) == 0;
241            }
242    
243            if ("<".equals(comparator)) {
244                return cal1.compareTo(cal2) < 0;
245            }
246    
247            if (">".equals(comparator)) {
248                return cal1.compareTo(cal2) > 0;
249            }
250    
251            if ("<=".equals(comparator)) {
252                return cal1.compareTo(cal2) <= 0;
253            }
254    
255            if (">=".equals(comparator)) {
256                return cal1.compareTo(cal2) >= 0;
257            }
258    
259            return false;
260        }

307         * Given a number of seconds, creates an int[] detailing the
308         * months, days, and hours difference as well.
309         * <p/>
310         * 0: days<br>
311         * 1: hours<br>
312         * 2:minutes<br>
313         * 3:seconds<br>

315         * @return int array

317        public static int[] dateArrayFromSeconds(int inSeconds) {
318            int days = (int) (inSeconds / (60 * 60 * 24));
319            inSeconds -= (days * 60 * 60 * 24);
320            int hours = (int) (inSeconds / (60 * 60));
321            inSeconds -= (hours * 60 * 60);
322            int minutes = (int) (inSeconds / 60);
323            inSeconds -= (minutes * 60);

375        /**
376         * Returns an int array detailing the difference between to Calendars <br>
377         * 0: days<br>
378         * 1: hours<br>
379         * 2:minutes<br>
380         * 3:seconds<br>
381         */
382        public static int[] diffMin(Calendar start, Calendar end) {
383            double numMinutes = ((double) (end.getTimeInMillis() - start.getTimeInMillis()) / 60000);
384            int days = (int) (numMinutes / 1440);
385            numMinutes -= days * 1440;
386            int hours = (int) (numMinutes / 60);
387            numMinutes -= hours * 60;
388            int minutes = (int) (numMinutes);
389            numMinutes -= minutes;
390            int seconds = (int) (numMinutes * 60);
391            return new int[]{days, hours, minutes, seconds};
392        }

467        /**
468         * Diffs the number of working seconds between two dates (takes into consideration a 7-5
469         * workday)
470         */
471        public static int diffWorkingSeconds(Calendar start, Calendar end) {
472            if (start == null || end == null) {
473                return 0;
474            }
475    
476            if (end.before(start)) {
477                Calendar newStart = end;
478                Calendar newEnd = start;
479                end = newEnd;
480                start = newStart;
481            }
482    
483            roundForward(start);
484            roundBack(end);
485    
486            if (!end.after(start)) {
487                return 0;
488            }
489    
490            int startDayOfWeek = start.get(Calendar.DAY_OF_WEEK);
491            int endDayOfWeek = end.get(Calendar.DAY_OF_WEEK);
492    
493            int weeksDiff = 0;
494            int daysDiff = end.get(Calendar.DAY_OF_YEAR) - start.get(Calendar.DAY_OF_YEAR);
495            if (daysDiff < 0) {
496                weeksDiff += 52;
497                daysDiff = Math.abs(daysDiff);
498            }
499            weeksDiff += daysDiff / 7;
500            if (endDayOfWeek < startDayOfWeek) {
501                weeksDiff++;
502            }
503    
504            int secondsDiff = diff(start.getTime(), end.getTime());
505            secondsDiff -= (((24 - getWorkDayLength()) * 60 * 60 * daysDiff) + ((2 * getWorkDayLength()) * weeksDiff * 60 * 60));
506            return secondsDiff;
507        }

537         * Returns a readable String based on diffMin(Calendar start,Calendar end)<BR>
538         * i.e. 3d 4h 2m 1s
539         *
540         * @see #diffMin(java.util.Calendar,java.util.Calendar)
541         */
542        public static String formatArray(int[] i) {
543            String results = "";
544            if (i[0] > 0) {
545                results += i[0] + "d ";
546            }
547            if (i[1] > 0) {
548                results += i[1] + "h ";
549            }
550            if (i[2] > 0) {
551                results += i[2] + "m ";
552            }
553            if (i[3] > 0) {
554                results += i[3] + "s ";
555            }
556            if (results.length() == 0) {
557                return "<1s";
558            }
559            return results;
560        }

991            }
992    
993            end.set(Calendar.HOUR_OF_DAY, 2);
994            compare.set(Calendar.HOUR_OF_DAY, 1);
995    
996            // after

1198        private static void buildDateFormats(String dateString, Set<StringBuilder> formats,
1199                                             String separator, String ddmmformat) {
1200            if (dateString.indexOf(separator) > 3) {
1201                formats.add(new StringBuilder().append("yyyy" + separator + ddmmformat));
1202            } else {
1203                int max = 0;
1204    
1205                //This is to figure out whether we are dealing with a 4 or 2 day year.
1206                //First, we try to split the string out by major separators (space and comma)
1207                String[] spaceParts = dateString.split(" ");
1208                for (String spacePart : spaceParts) {
1209                    final String[] commaParts = spacePart.split(",");
1210                    for (String commaPart : commaParts) {
1211                        //Count the number of separators to detect if we're looking at the date or the time
1212                        final int numSeparator = StringUtils.count(commaPart, separator.charAt(0));
1213                        if (numSeparator > 1) {
1214                            String[] dateParts = commaPart.split(separator);
1215    
1216                            for (String datePart : dateParts) {
1217                                max = Math.max(max, datePart.length());
1218                            }
1219                        }
1220                    }
1221                }
1222    
1223                if(max == 0) {
1224                    max = 4;
1225                }
1226    
1227                if (max > 2) {
1228                    formats.add(new StringBuilder().append(ddmmformat).append(separator).append("yyyy"));
1229                } else {
1230                    formats.add(new StringBuilder().append(ddmmformat).append(separator).append("yy"));
1231                }
1232    
1233            }
1234        }

1249        /**
1250         * Converts a date in the form of a String to a Date object. Checks for tokens such as '/', ':', and '-' to help
1251         * separate out the month, day, and year.
1252         *
1253         * @param str date string
1254         * @return parsed Date
1255         */
1256        public static Date parseDate(String str) {
1257            if (!StringUtils.is(str)) {
1258                return null;
1259            }
1260    
1261            Set<StringBuilder> formats = new LinkedHashSet<StringBuilder>();
1262    
1263            //Start with dates
1264            if (str.indexOf("/") > -1) {
1265                buildDateFormats(str, formats, "/", MMDD_SLASH_FORMAT);
1266            } else if (str.indexOf("-") > -1) {
1267                buildDateFormats(str, formats, "-", MMDD_DASH_FORMAT);
1268            } else {
1269                formats.add(new StringBuilder().append("MMM dd yyyy"));
1270                formats.add(new StringBuilder().append("MMM dd, yyyy"));
1271                formats.add(new StringBuilder().append("MMM dd, yyyy"));
1272                formats.add(new StringBuilder().append("yyyyMMdd"));
1273                formats.add(new StringBuilder().append("MMdd"));
1274                formats.add(new StringBuilder().append(""));
1275            }
1276    
1277            //This means we have 'time' of some sort
1278            int count = StringUtils.count(str, ':');
1279            if (count > 0) {
1280                if (str.indexOf(",") > -1) {
1281                    appendFormats(formats, true, ",", " ");
1282                } else {
1283                    appendFormats(formats, false, " ");
1284                }
1285    
1286                final int indexOfColon = str.indexOf(":");
1287                int amIndex = str.toLowerCase().indexOf("am");
1288                int pmIndex = str.toLowerCase().indexOf("pm");
1289    
1290                final boolean hasAMPM = amIndex > indexOfColon || pmIndex > indexOfColon;
1291                if (hasAMPM) {
1292                    appendFormats(formats, false, "hh");
1293                } else {
1294                    appendFormats(formats, false, "HH");
1295                }
1296    
1297                appendFormats(formats, false, ":mm");
1298                if (count > 1) {
1299                    appendFormats(formats, false, ":ss");
1300                }
1301    
1302                if (hasAMPM) {
1303                    appendFormats(formats, false, " aa");
1304                }
1305            }
1306    
1307            //This one was already in there
1308            formats.add(new StringBuilder().append("E MMM dd HH:mm:ss zzz yyyy"));
1309            Date parsed = null;
1310            formats:
1311            for (StringBuilder format : formats) {
1312                try {
1313                    DateFormat dateFormat = new SimpleDateFormat(format.toString());
1314                    parsed = dateFormat.parse(str);
1315                    break formats;
1316                } catch (Exception e) {
1317                    System.out.println("Unable to parse date: " + str + StringUtils.parent(format));
1318                }
1319            }
1320            return parsed;
1321        }

1325         * existing formats
1326         *
1327         * @param formats
1328         * @param toAppend
1329         */
1330        private static void appendFormats(Set<StringBuilder> formats, boolean create, String... toAppend
1331        ) {

1339                        thisCreate = false;
1340                    } else {
1341                        format.append(itemToAppend);
1342                    }
1343                }
1344            }
1345            if (create && toAdd != null) {
1346                formats.addAll(toAdd);
1347            }
1348        }

1601        public static int getServerOffset(boolean useDaylightSavingTime) {
1602            boolean observingDaylightSavingTime = false;
1603            final int DST = 1;        
1604    
1605            // New calendar to use as a reference point.
1606            Calendar today = new GregorianCalendar();
1607    
1608            // TimeZone set to 'EST' for GMT comparison so we can determine the offset.
1609            TimeZone serverTimeZone = TimeZone.getTimeZone("America/New_York");
1610    
1611            // First getTime() returns date, the second getTime() converts that to milliseconds since the default date in java.
1612            int offsetMilliseconds = serverTimeZone.getOffset(today.getTime().getTime());
1613    
1614            // 3600000 = 1000 (ms in a sec) * 60 (sec in a min) * 60 (min in a hour)
1615            int offsetHours = (offsetMilliseconds / 3600000);
1616    
1617            // If server is on DST since our servers are EDT
1618            if(offsetHours == -4) {
1619                observingDaylightSavingTime = true;
1620            }
1621    
1622            if(observingDaylightSavingTime && useDaylightSavingTime) {
1623                offsetHours -= DST;
1624            }
1625    
1626            return offsetHours;
1627        }
