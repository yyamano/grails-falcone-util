001    package com.infusion.util;
002    
003    import java.io.Serializable;
004    
005    /**
006     * A convenience class that makes it easy to return multiple objects in a type-safe way
007     */
008    public class Duple<T, U> implements Serializable {
009    // ========================================================================================================================
010    //    Instance Fields
011    // ========================================================================================================================
012    
013        public final T f1;
014        public final U f2;
015    
016    // ========================================================================================================================
017    //    Constructors
018    // ========================================================================================================================
019    
020        public Duple(T f1, U f2) {
021            this.f1 = f1;
022            this.f2 = f2;
023        }
024    
025    // ========================================================================================================================
026    //    Public Instance Methods
027    // ========================================================================================================================
028    
029        public boolean equals(Object o) {
030            if (this == o) return true;
031            if (o == null || getClass() != o.getClass()) return false;
032    
033            final Duple duple = (Duple) o;
034    
035            if (f1 != null ? !f1.equals(duple.f1) : duple.f1 != null) return false;
036            if (f2 != null ? !f2.equals(duple.f2) : duple.f2 != null) return false;
037    
038            return true;
039        }
040    
041        public int hashCode() {
042            int result;
043            result = (f1 != null ? f1.hashCode() : 0);
044            result = 29 * result + (f2 != null ? f2.hashCode() : 0);
045            return result;
046        }
047    
048        public String toString() {
049            return f1 + ", " + f2;
050        }
051    }
