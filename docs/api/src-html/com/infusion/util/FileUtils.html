028         * Opens a file (targetFile) and appends a String (toWrite) to it
029         */
030        public static void appendFile(File targetFile, String toWrite) throws Exception {
031            if (!targetFile.exists()) {
032                if (!targetFile.createNewFile()) throw new Exception("File was not created!");
033            }
034            if (!targetFile.canWrite())
035                throw new Exception("No permission to write file!");
036            else {
037                try {
038                    // WE CAN WRITE TO THE FILE
039                    FileWriter fw = new FileWriter(targetFile, true);
040                    fw.write(toWrite);
041                    fw.close();
042                } catch (Exception e) {
043                    throw e;
044                }
045            }
046        }

058        /**
059         * Copies an inputstream to an output stream
060         *
061         * @param in
062         * @param out
063         * @param bufferSize
064         * @throws IOException
065         */
066        public static void copyBuffer(BufferedInputStream in, BufferedOutputStream out, int bufferSize) throws IOException {
067            while (in.available() > 0) {
068                out.write(in.read());
069            }
070            in.close();
071            out.flush();
072            out.close();
073        }

075        /**
076         * Copies an inputstream to an output stream
077         *
078         * @param in
079         * @param out
080         * @param bufferSize
081         * @throws IOException
082         */
083        public static void copyBuffer(InputStream in, OutputStream out, int bufferSize) throws IOException {
084    
085            byte[] buffer = new byte[bufferSize];
086    
087            try {
088    
089                while (true) {
090                    int amountRead = in.read(buffer);
091                    if (amountRead == -1) {
092                        break;
093                    }
094                    out.write(buffer, 0, amountRead);
095                }
096            } finally {
097                if (in != null) {
098                    in.close();
099                }
100                if (out != null) {
101                    out.close();
102                }
103            }
104        }

119        /**
120         * Copies from one file to another
121         *
122         * @param s source file
123         * @param d destination file
124         * @throws IOException
125         */
126        public static void copyFile(File s, File d) throws IOException {
127            if (!s.canRead()) throw new IOException("Unable to read source file.");
128            if (d.exists() && !d.canWrite()) throw new IOException("Unable to write dest file");
129            if (!d.exists() && !d.createNewFile()) throw new IOException("Unable to create new file");
130    
131            BufferedInputStream is = new BufferedInputStream(new FileInputStream(s));
132            BufferedOutputStream os = new BufferedOutputStream(new FileOutputStream(d));
133            byte[] b = new byte[1024];
134            int amountRead = is.read(b);
135    
136            while (amountRead > 0) {
137                os.write(b, 0, amountRead);
138                amountRead = is.read(b);
139            }
140            os.close();
141            is.close();
142        }

176        /**

178         *
179         * @param file
180         * @param exts Extension list to compare against
181         */
182        public static boolean isType(File file, String... exts) {
183            if (file == null) return false;
184            for (String extension : exts) {
185                String lowerFile = file.getName().toLowerCase();
186                if (lowerFile.endsWith(extension.toLowerCase())) {
187                    return true;
188                }
189            }
190            return false;
191        }

301         */
302        public static void readWrite(File f, OutputStream out) throws IOException {
303            if (!f.exists()) return;
304            if (!f.canRead() || !f.canWrite()) throw new IOException("Can't read or write template file.");
305    
306            FileInputStream input = new FileInputStream(f);
307            BufferedInputStream bis = new BufferedInputStream(input);
308    
309            BufferedOutputStream bos = new BufferedOutputStream(out);
310    
311    
312            int aByte;
313            while ((aByte = bis.read()) != -1) {
314                bos.write(aByte);
315            }
316            bos.flush();
317            bos.close();
318            bis.close();
319        }

337         * Given a raw size of a file, returns a readable size for it, ie:
338         * <p/>
339         * 1MB <BR>
340         * 12KB <BR>
341         * 30 bytes <BR>
342         * 30GB<BR>
343         *
344         * @param i File size
345         */
346        public static String readableSize(long i) {
347            if (i < 0) return "N/A";
348            if (i > 999999999) {
349                return NumberUtils.doubleFormat((double) i / 1000000000) + " GB";
350            }
351    
352            if (i > 999999) {
353                return NumberUtils.doubleFormat((double) i / 1000000) + " MB";
354            } else if (i > 999) {
355                return (int) (i / 1000) + " KB";
356            } else {
357                return i + " Bytes";
358            }
359        }

373         *
374         * @param file
375         * @param deleteParentFile Whether or not to delete the top level
376         */
377        public static void recursiveDelete(File file, boolean deleteParentFile) {
378            File[] files = file.listFiles();
379            for (int i = 0; i < files.length; i++) {
380                File file1 = files[i];
381                if (file1.isDirectory()) {
382                    recursiveDelete(file1);
383                } else {
384                    file1.delete();
385                }
386            }
387            if (deleteParentFile) {
388                file.delete();
389            }
390        }

425        /**
426         * Simple method to zip up a file
427         *
428         * @param source The file to be zipped
429         * @param target The target zip file
430         */
431        public static final void zipFile(File source, File target) throws IOException {
432    
433            FileOutputStream fo = new FileOutputStream(target);
434            ZipOutputStream zo = new ZipOutputStream(fo);
435            zo.setMethod(ZipOutputStream.DEFLATED);
436    
437            FileInputStream fi = new FileInputStream(source);
438            BufferedInputStream bi = new BufferedInputStream(fi);
439            ZipEntry ze = new ZipEntry(source.getName());
440            zo.putNextEntry(ze);
441    
442            byte[] dt = new byte[1024];
443            int bCnt;
444            while ((bCnt = bi.read(dt, 0, 1024)) != -1) {
445                zo.write(dt, 0, bCnt);
446            }
447            zo.flush();
448            zo.closeEntry();
449            bi.close();
450    
451            zo.close();
452            source.delete();
453        }

455        /**
456         * Creates a path given a list of path items, uses File.separator
457         *
458         * @param prefix Whether to prefix with separator
459         * @param suffix Whether to suffix with separator
460         * @param parts  The parts
461         * @return A String
462         */
463        public static String createPath(boolean prefix, boolean suffix, String... parts) {
464            StringBuilder path = new StringBuilder();
465    
466            for (int i = 0; i < parts.length; i++) {
467                String part = parts[i];
468                if (!part.startsWith(File.separator)) {
469                    if (i == 0 && prefix) {
470                        path.append(File.separator);
471                    }
472                }
473    
474                path.append(part);
475    
476                if (!part.endsWith(File.separator)) {
477                    if (i < (parts.length - 1) || suffix) {
478                        path.append(File.separator);
479                    }
480                }
481            }
482            return path.toString();
483        }
