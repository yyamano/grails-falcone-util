001    package com.infusion.util;
002    
003    
004    import java.io.*;
005    import java.util.LinkedHashMap;
006    import java.util.Map;
007    
008    /**
009     * Utilities for dealing with maps.
010     */
011    
012    public abstract class MapUtils {
013    // ========================================================================================================================
014    //    Static Methods
015    // ========================================================================================================================
016    
017        /**
018         * Inverts this filter (the keys become the values, and the values become the keys).
019         */
020    
021        public static <T> Map<T, T> invert(Map<T, T> source) {
022            return invert(source, new LinkedHashMap<T, T>());
023        }
024    
025        /**
026         * Inverts this filter (the keys become the values, and the values become the keys).  New values get put into target.
027         */
028    
029        public static <T> Map<T, T> invert(Map<T, T> source, Map<T, T> target) {
030            for (Map.Entry<T, T> entry : source.entrySet()) {
031                target.put(entry.getValue(), entry.getKey());
032            }
033    
034            return target;
035        }
036    
037        public static Map toMap(Reader reader) throws IOException {
038            BufferedReader r = new BufferedReader(reader);
039            Map map;
040            String line = r.readLine();
041            map = new LinkedHashMap();
042            while (line != null) {
043                if (!"".equals(line)) {
044                    if (line.charAt(0) != '#') {
045                        String[] piece = line.split(",");
046                        if (piece.length == 1) {
047                            map.put(piece[0], piece[0]);
048                        } else if (piece.length > 1) {
049                            map.put(piece[0], piece[1]);
050                        } else if (piece.length == 0) {
051                            throw new InternalError("Invalid syntax in file: " + line);
052                        }
053                    }
054                }
055                line = r.readLine();
056            }
057            return map;
058        }
059    }
