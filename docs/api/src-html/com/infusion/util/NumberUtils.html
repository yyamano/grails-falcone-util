067        /**
068         * Given an integer returns the number with the appropriate appreviation after it.
069         * (ie 1st, 2nd, 24th)
070         *
071         * @param number some number
072         * @return string with number and appreviation
073         */
074        public static String Nd(int number) {
075            String n = String.valueOf(number);
076            char c = n.charAt(n.length() - 1);
077            if (n.length() > 1) {
078                char t = n.charAt(n.length() - 2);
079                if (t == '1') {
080                    return number + "th";
081                }
082            }
083            switch (c) {
084                case'1':
085                    return number + "st";
086                case'2':
087                    return number + "nd";
088                case'3':
089                    return number + "rd";
090                default:
091                    return number + "th";
092            }
093        }

166        public static String formatDollar(Double d, Locale locale) {
167            if (d == null) {
168                return null;
169            }
170    
171            DecimalFormat df = (DecimalFormat) DecimalFormat.getCurrencyInstance(locale);
172            String f = df.format(d);
173            f = f.replaceAll(" ", "");// for some reason South African Rands add a space
174    
175            if (f.indexOf("(") > -1) {
176                f = "-" + f.replaceAll("\\(|\\)", "");
177            }
178    
179            // todo: find a better solution
180            // broswers don't like it for some reason - unless you do this, the £ shows as a ?
181    //        if (f.contains("£")) {
182    //            f = f.replace("£", "&pound;");
183    //        }
184    
185            return f;
186        }

188        /**
189         * Formats an amount depending on the locale. Gets rid of trailing decimals only if they are zeros and the flag is
190         * set.
191         *
192         * @param amount double amount
193         * @param locale country format to display the amount in
194         * @param includeZeros whether to keep trailing zeros in the decimal or not (ie .00)
195         * @return currency string of amount given in the format desired by a locale
196         */
197        public static String formatCurrency(double amount, Locale locale, boolean includeZeros) {
198            String currencyString = formatDollar(amount, locale);
199            if (!includeZeros) {
200                currencyString = currencyString.replaceAll("(\\.0$)|(\\.00$)", "");
201            }
202            return currencyString;
203        }

290            if (a == null) return b;
291            if (b == null) return a;
292            if (a.doubleValue() > b.doubleValue()) {
293                return b;
294            } else {
295                return a;

334         * Parses a double, accepts null values
335         */
336        public static double parseDouble(String str) {
337            if (!StringUtils.is(str)) {
338                return 0.0;
339            } else {
340                String stripString = StringUtils.stripNonNumber(str);
341                if (!StringUtils.is(stripString)) {
342                    return 0.0;
343                } else {
344                    return Double.parseDouble(stripString);
345                }
346            }
347        }

365        /**
366         * Parses an int, accepts null values
367         *
368         * @param str input string to parse
369         * @return integer version of string. Returns 0 if the string is null
370         */
371        public static int parseInt(String str) {
372            if (str == null) return 0;
373    
374            //Check for simple y/n
375            if (str.length() == 1 && Character.isLetter(str.charAt(0))) {
376                return "y".equalsIgnoreCase(str) ? 1 : 0;
377            }
378            // check if the number is in scientific notation
379    //        if (str.contains("E") && str.charAt(1) == '.') {
380    //            str = str.substring(0, str.indexOf("E")).replaceAll("\\.", "");
381    //        }
382            str = StringUtils.stripNonNumber(str);
383            if (!StringUtils.is(str))
384                return 0;
385            else
386                return new Double(str).intValue();
387        }

406        public static int randomInt(int ceil) {
407            Random m = new Random(Calendar.getInstance().getTimeInMillis());
408            Random e = new Random(m.nextInt());
409            return (int) (e.nextFloat() * (ceil + 1));
