393        /**
394         * Returns whatever is between the < > in the String
395         *
396         * @param email email to parse
397         * @return parsed email
398         */
399    
400        public static String ParseEmail(String email) {
401            String parsed = null;
402    
403            if (email != null) {
404                int start = 0;
405                int stop = 0;
406                if ((email.contains("<")) && (email.contains(">"))) {
407                    start = email.lastIndexOf("<") + 1;
408                    stop = email.lastIndexOf(">");
409                } else if ((email.contains("&lt;")) && (email.contains(">"))) {
410                    start = email.lastIndexOf("&lt;") + 4;
411                    stop = email.lastIndexOf(">");
412                }
413                if (start < stop) {
414                    parsed = email.substring(start, stop);
415                } else {
416                    parsed = email;
417                }
418            }
419    
420            if (parsed != null) {
421                return parsed.replace("<", "").replace("&lt;", "").replace(">", "").replace("\"", "").trim().replace(" ", "");
422            } else {
423                return "";
424            }
425        }

724        /**
725         * Wraps text to a certain width
726         */
727        public static String blockText(String textToBlock, int blockwidth) {
728            StringBuffer returnVal = new StringBuffer();
729            StringBuffer tmp = new StringBuffer();
730            if (textToBlock == null || textToBlock == "" || blockwidth == 0) {
731                return null;
732            } else {
733                returnVal = new StringBuffer();
734                tmp = new StringBuffer(textToBlock);
735                int curr = 0;
736                for (int i = 0; i < tmp.length(); i++) {
737                    curr++;
738                    char c = tmp.charAt(i);
739                    if (c == '\n') {
740                        curr = 0;
741                    }
742                    if (curr >= (blockwidth - 4)) {
743                        if (c == ' ' || curr >= blockwidth + 6) {
744                            returnVal.append('\n');
745                            //tmp.insert(i+1, '\n');
746                            while (c == ' ') {
747                                if (tmp.length() <= i) {
748                                    break;
749                                }
750                                c = tmp.charAt(i);
751                                i++;
752                            }
753                            i--;
754                            curr = 0;
755                        }
756                    }
757                    returnVal.append(c);
758                }
759                return returnVal.toString();
760            }
761        }
