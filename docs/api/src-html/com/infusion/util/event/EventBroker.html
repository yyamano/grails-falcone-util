001    package com.infusion.util.event;
002    
003    /**
004     * Iterface for defining an event broker - an object to which people subscribe to events and publish events.
005     */
006    public interface EventBroker {
007    // ========================================================================================================================
008    //    Public Instance Methods
009    // ========================================================================================================================
010    
011        /**
012         * Returns an eventType definition with a given name
013         */
014        EventType getEventType(String eventName);
015    
016        /**
017         * Loads a bunch of config from an eventConfiguration instance
018         *
019         * @param eventConfiguration
020         */
021        void loadConfiguration(EventConfiguration eventConfiguration);
022    
023        /**
024         * Public method - this is the entry point to fire an event.  This class will pick up top-level consumers for the event
025         * and use the notifiers to notify lower level events.
026         *
027         * @param event THe event being thrown
028         */
029        void publish(Object event);
030    
031        /**
032         * Internal method for calling an event, looks up notifiers and calls them appropriately.
033         *
034         * @param event     The original event being fired
035         * @param eventName Which event class we are currently examining
036         */
037        void publish(String eventName, Object event);
038    
039        /**
040         * Registers a new event type
041         */
042        void registerEventType(EventType eventType);
043    
044        /**
045         * Removes a bunch of config from an eventConfiguration instance
046         *
047         * @param eventConfiguration
048         */
049        void removeConfiguration(EventConfiguration eventConfiguration);
050    
051        /**
052         * Removes an event type definition.
053         *
054         * @param eventType
055         */
056        void removeEventType(EventType eventType);
057    
058        /**
059         * This adds an item to the consumers map.
060         *
061         * @param eventType The event class this will be fired on
062         * @param consumer  The services to handle that event (generics shuold enforce the contract)
063         */
064        void subscribe(String eventType, EventConsumer consumer);
065    
066        /**
067         * Removes a consumer from listening ona given event.
068         */
069        boolean unsubscribe(String event, EventConsumer consumer);
070    }
