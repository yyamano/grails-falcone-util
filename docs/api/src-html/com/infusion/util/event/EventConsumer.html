001    package com.infusion.util.event;
002    
003    /**
004     * A class that will handle an event of a particular type.  The Type parameter E is the type
005     * of event this record will service.
006     *
007     * @author eric
008     */
009    public interface EventConsumer<E> {
010    // ========================================================================================================================
011    //    Public Instance Methods
012    // ========================================================================================================================
013    
014        /**
015         * This method will perform the work needed on the event, if any.  Because we can have a heirarchy of
016         * events, it also passes in the event class that is calling this method.
017         *
018         * @param event The event that was fired
019         * @param eventBroker The class that defined this handler
020         */
021        public void consume(E event, EventBroker eventBroker);
022    }
