001    package com.infusion.util.event;
002    
003    /**
004     * A delegate is passed into the eventBroker as an event, but defers the creation of the
005     * event object until the broker knows that there are people that actually care about the
006     * event.  This could be used when:
007     *
008     * <ul>
009     * <li>The creation of the event object is expensive</li>
010     * <li>There is work needed to transform one event to another form and you don't want to spend cycles when there's
011     * no one that really cares</li>
012     * </ul>
013     */
014    public interface EventDelegate {
015    // ========================================================================================================================
016    //    Public Instance Methods
017    // ========================================================================================================================
018    
019        /**
020         * Returns the actual event
021         * @return
022         */
023        public Object getEvent(EventBroker eventBroker);
024    }
