001    package com.infusion.util.event;
002    
003    /**
004     * An exception thrown when dealing with events.
005     */
006    public class EventException extends Exception {
007        public EventException() {
008            super();
009        }
010    
011        public EventException(String s) {
012            super(s);
013        }
014    
015        public EventException(String s, Throwable throwable) {
016            super(s, throwable);
017        }
018    
019        public EventException(Throwable throwable) {
020            super(throwable);
021        }
022    }
