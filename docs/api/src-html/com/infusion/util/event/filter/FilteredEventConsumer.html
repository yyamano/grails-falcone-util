001    package com.infusion.util.event.filter;
002    
003    import com.infusion.util.event.filter.EventFilter;
004    import com.infusion.util.event.EventConsumer;
005    import com.infusion.util.event.EventBroker;
006    
007    import static java.text.MessageFormat.*;
008    
009    /**
010     * Created by IntelliJ IDEA.

015    public class FilteredEventConsumer<E> implements EventConsumer<E> {
016        private final EventConsumer<E> consumer;
017        private final EventFilter<E> filter;
018    
019        public FilteredEventConsumer(EventConsumer<E> consumer, EventFilter<E> filter) {
020            this.consumer = consumer;
021            this.filter = filter;
022        }
023    
024        public void consume(E event, EventBroker eventBroker) {
025            if (filter.filter(event)) {
026                consumer.consume(event, eventBroker);
027            }
028        }
029    
030        public String toString() {
031            return format("Filtered Consumer Wrapper: {0}", consumer);
032        }
033    }
