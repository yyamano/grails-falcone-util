001    package com.infusion.util.event.groovy;
002    
003    import com.infusion.util.event.chain.ChainedEventConsumer;
004    import com.infusion.util.event.chain.EventTransformer;
005    import com.infusion.util.event.EventType;
006    import com.infusion.util.event.EventException;
007    import groovy.lang.Closure;
008    
009    /**
010     * An extension of ChainedEventConsumer that allows to pass closures as parameters.
011     */
012    public class GroovyChainedEventConsumer extends ChainedEventConsumer {
013    // ========================================================================================================================
014    //    Constructors
015    // ========================================================================================================================
016    
017         /**
018         * Constructor that takes in a closure for the transformer.
019         *
020         * @param source
021         * @param target
022         * @param transformer
023         */
024        public GroovyChainedEventConsumer(String source, String target, final Closure transformer) {
025            super(source, target);
026            this.targetEvent = target;
027            this.eventTransformer = new EventTransformer() {
028                public Object transform(Object source) {
029                    return transformer.call(source);
030                }
031            };
032        }
033    
034        public GroovyChainedEventConsumer(EventType source,
035                                    EventType target, final Closure transformer) throws EventException {
036            super(source, target);
037            this.eventTransformer = new EventTransformer() {
038                public Object transform(Object source) {
039                    return transformer.call(source);
040                }
041            };
042        }
043    }
