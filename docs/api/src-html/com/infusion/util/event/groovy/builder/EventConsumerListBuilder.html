001    package com.infusion.util.event.groovy.builder;

003    import groovy.lang.GroovyObjectSupport;
004    import groovy.lang.Closure;
005    
006    import java.util.Map;
007    import java.util.HashMap;
008    
009    import com.infusion.util.event.EventConsumer;
010    
011

018    public class EventConsumerListBuilder extends GroovyObjectSupport {
019        Map<EventConsumer, String> eventConsumers = new HashMap<EventConsumer, String>();
020    
021        @Override
022        public Object invokeMethod(String s, Object o) {
023            EventConsumerBuilder eventConsumerBuilder = new EventConsumerBuilder(s, this);
024            final Object[] args = (Object[]) o;
025            Closure c = (Closure) args[0];
026            c.setResolveStrategy(Closure.DELEGATE_FIRST);
027            c.setDelegate(eventConsumerBuilder);
028            c.call();
029            return eventConsumerBuilder;
030        }
031    
032        public Map<EventConsumer, String> getEventConsumers() {
033            return eventConsumers;
034        }
035    }
