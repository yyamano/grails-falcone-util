001    package com.infusion.util.event.groovy.builder;
002    
003    import groovy.lang.GroovyObjectSupport;
004    
005    import java.util.List;
006    import java.util.ArrayList;
007    
008    import com.infusion.util.event.EventType;
009    
010    /**
011     * EventBuilder is a class used to build EventType objects.  You use it like this:
012     *
013     * <pre>
014     * def b = new EventTypeBuilder();
015     * b {
016     *     myEventName(MyEventClass)
017     *     anotherEventName(AnotherEventClass)
018     * }
019     * </pre>
020     */
021    public class EventTypeListBuilder extends GroovyObjectSupport {
022    
023        List<EventType> eventTypes = new ArrayList<EventType>();
024    
025        @Override
026        public Object invokeMethod(String name, Object o) {
027            Object[] args = (Object[]) o;
028            eventTypes.add(new EventType((Class) args[0], name));
029            return this;
030        }
031    
032        public List<EventType> getEventTypes() {
033            return eventTypes;
034        }
035    }
