011    public class IndexedTreeMap<K, I> implements Map<K, Object> {
012        private TreeMap<K, Object> internal;
013        private Map<I, K> keyIndex;
014        private TreeMapIndexer treeMapIndexer;
015    
016        public IndexedTreeMap(Comparator comparator, TreeMapIndexer<K, I> indexer) {
017            this.internal = new TreeMap<K, Object>(comparator);
018            this.keyIndex = new HashMap<I, K>();
019            this.treeMapIndexer = indexer;
020        }
021    
022        public boolean containsValue(Object o) {
023            return internal.containsValue(o);
024        }
025    
026        public Object get(Object o) {
027            return internal.get(o);
028        }
029    
030        public Comparator<? super K> comparator() {
031            return internal.comparator();
032        }
033    
034        public K firstKey() {
035            return internal.firstKey();
036        }
037    
038        public K lastKey() {
039            return internal.lastKey();
040        }
041    
042        public void putAll(Map<? extends K, ? extends Object> map) {
043            internal.putAll(map);
044        }
045    
046        public Object put(K k, Object o) {
047            if(treeMapIndexer != null) {
048                treeMapIndexer.index(k, keyIndex);
049            }
050            return internal.put(k, o);
051        }
052    
053        public Object remove(Object o) {
054            if(treeMapIndexer != null) {
055                treeMapIndexer.remove(o, keyIndex);
056            }
057            return internal.remove(o);
058        }
059    
060        public void clear() {
061            internal.clear();
062            keyIndex.clear();
063        }
064    
065        public Object clone() {
066            return internal.clone();
067        }
068    
069        public Set<K> keySet() {
070            return internal.keySet();
071        }
072    
073        public Collection<Object> values() {
074            return internal.values();
075        }
076    
077        public Set<Entry<K, Object>> entrySet() {
078            return internal.entrySet();
079        }
080    
081        public SortedMap<K, Object> subMap(K k, K k1) {
082            return internal.subMap(k, k1);
083        }
084    
085        public SortedMap<K, Object> headMap(K k) {
086            return internal.headMap(k);
087        }
088    
089        public SortedMap<K, Object> tailMap(K k) {
090            return internal.tailMap(k);
091        }
092    
093        public boolean isEmpty() {
094            return internal.isEmpty();
095        }
096    
097        public boolean equals(Object o) {
098            return internal.equals(o);
099        }
100    
101        public int hashCode() {
102            return internal.hashCode();
103        }
104    
105        public String toString() {
106            return internal.toString();
107        }
108    
109        public int size() {
110            return internal.size();
111        }
112    
113        public boolean containsKey(Object o) {
114            return internal.containsKey(o);
115        }
116    
117        public Object getByIndex(I i) {
118            return get(keyIndex.get(i));
119        }
120    
121        public boolean containsKeyByIndex(I i) {
122            return containsKey(keyIndex.get(i));
123        }
124    
125        public Object removeByIndex(I i) {
126            return remove(keyIndex.get(i));
127        }
128    
129    }
